# ex15 (ex da soma, media total de um aluno)
notas =[7.5 , 8.0 , 6.5 , 9.0 , 7.0]

def calcular_media(notas) :
    total = sum(notas)
    media = total/len(notas)
    return media
arredondar_media = lambda media: round(media,2)

media = calcular_media(notas)
media_arredondada = arredondar_media(media)

if media_arredondada >= 7:
    situacao = print("Aprovado")
else:
    situacao = print("reprovado")

print(notas)
print(media_arredondada)
print(situacao)

#ex16 (ex da compra de um produto com valor e desconto somados)
valor_do_produto = int(input("Digite o valor do produto R$: "))
desconto = int(input("Digite o valor do desconto%: "))

if desconto < 0 or desconto > 100:
    print("ERRO: O desconto deve estar entre 0% e 100%")
else:

    valor_desconto = valor_do_produto * (desconto / 100)
    total_com_desconto = valor_do_produto - valor_desconto

    print(f"Valor do desconto: R$ {valor_desconto:.2f}")
    print(f"Total a pagar: R$ {total_com_desconto:.2f}")
